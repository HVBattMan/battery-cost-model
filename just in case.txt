CREATE TABLE `cell_input` (
  `cellId` int(11) NOT NULL,
  `cellName` varchar(45) NOT NULL,
  `cellType` varchar(45) NOT NULL,
  `npRatio` double NOT NULL,
  `electrodeLength` double NOT NULL,
  `electrodeWidth` double NOT NULL,
  `catFormulaId` int(11) DEFAULT NULL,
  `catGravCapacity` double NOT NULL,
  `catTotalLoading` double NOT NULL,
  `catActiveFrac` double NOT NULL,
  `catBinderFrac` double NOT NULL,
  `catConductorFrac` double NOT NULL,
  `anGravCapacity` double NOT NULL,
  `anActiveFrac` double NOT NULL,
  `anBinderFrac` double NOT NULL,
  `anConductorFrac` double NOT NULL,
  `alFoilThickness` double NOT NULL,
  `cuFoilThickness` double NOT NULL,
  `elyte` varchar(45) DEFAULT NULL,
  `separatorName` varchar(45) DEFAULT NULL,
  `can` varchar(45) DEFAULT NULL,
  `avgDischargeVoltage` double NOT NULL,
  PRIMARY KEY (`cellId`),
  UNIQUE KEY `cell_id_UNIQUE` (`cellId`),
  UNIQUE KEY `cell_name_UNIQUE` (`cellName`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

  `anConductor` double DEFAULT NULL,
  `alFoil` double DEFAULT NULL,
  `cuFoil` double DEFAULT NULL,
  `can` double DEFAULT NULL,

  `elyte` double DEFAULT NULL,
